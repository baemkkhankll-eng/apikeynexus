บริการตรวจสอบ HWID Key Authentication

กลับไปแดชบอร์ด

ภาพรวม

API นี้ช่วยให้คุณสามารถตรวจสอบคีย์ลิขสิทธิ์ที่ล็อกด้วย HWID ในแอปพลิเคชันของคุณ แต่ละคีย์จะเชื่อมโยงกับ Developer ID เฉพาะและสามารถใช้ได้บนเครื่องเดียวเท่านั้น (HWID)

Base URL: http://localhost:3000

การใช้งาน Python

การติดตั้ง

pip install requests

รับ HWID ของเครื่อง

import uuid
import hashlib

def get_hwid():
    mac = uuid.getnode()
    hwid = hashlib.sha256(str(mac).encode()).hexdigest()[:32]
    return hwid

const { machineIdSync } = require('node-machine-id');
const crypto = require('crypto');

function getHWID() {
  const id = machineIdSync();
  return crypto.createHash('sha256').update(id).digest('hex').substring(0, 32);
}

ตัวอย่างแบบเต็ม

const { machineIdSync } = require('node-machine-id');
const crypto = require('crypto');
const axios = require('axios');

class HWIDAuth {
  constructor(apiUrl, developerId) {
    this.apiUrl = apiUrl;
    this.developerId = developerId;
  }
  
  getHWID() {
    const id = machineIdSync();
    return crypto.createHash('sha256').update(id).digest('hex').substring(0, 32);
  }
  
  async validate(key) {
    const hwid = this.getHWID();
    try {
      const response = await axios.post(`${this.apiUrl}/api/validate`, {
        key: key,
        hwid: hwid,
        developerId: this.developerId
      });
      return response.data;
    } catch (error) {
      return { valid: false, message: error.message };
    }
  }
}

// Usage
const auth = new HWIDAuth("http://localhost:3000", "YOUR_DEVELOPER_ID");

async function startApp() {
  const result = await auth.validate("KEY-XXXX-XXXX-XXXX-XXXX");
  
  if (result.valid) {
    console.log("License valid! Application starting...");
    // Start your application
  } else {
    console.log(`License invalid: ${result.message}`);
    process.exit(1);
  }
}

startApp();

อ้างอิง API

POST /api/validate

ตรวจสอบคีย์ลิขสิทธิ์พร้อมการตรวจสอบ HWID

Request Body

{
  "key": "KEY-XXXX-XXXX-XXXX-XXXX",
  "hwid": "32-character-hwid-hash",
  "developerId": "DEV-XXXXXXXX"
}

Response (สำเร็จ)

{
  "valid": true,
  "message": "Key valid"
}

Response (ข้อผิดพลาด)

{
  "valid": false,
  "message": "Key is banned"
}

ข้อความข้อผิดพลาด

  • Invalid key - คีย์ไม่มีอยู่จริง
  • Key is banned - คีย์ถูกระงับโดยเจ้าของ
  • Key expired - คีย์หมดอายุแล้ว
  • HWID mismatch - คีย์ถูกล็อกกับเครื่องอื่น
  • Developer ID mismatch - Developer ID ไม่ถูกต้อง

หมายเหตุสำคัญ

  • • รับ Developer ID ของคุณจากแดชบอร์ดหลังจากเข้าสู่ระบบด้วย Discord
  • • HWID ควรคงที่เมื่อเริ่มแอปพลิเคชันใหม่บนเครื่องเดียวกัน
  • • คีย์จะถูกล็อกกับเครื่องแรกที่ใช้งาน (HWID locking)
  • • หากผู้ใช้เปลี่ยนฮาร์ดแวร์ ให้ใช้ฟีเจอร์ "รีเซ็ต HWID" ในแดชบอร์ด
  • • เก็บ Developer ID ของคุณให้ปลอดภัย - จำเป็นสำหรับการเรียก API ทั้งหมด
  • • สำหรับการใช้งานจริง ให้ใช้ HTTPS แทน HTTP
